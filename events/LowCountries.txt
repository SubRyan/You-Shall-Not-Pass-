### The Fall of Belgium

# The Battle is Lost
country_event = {
	id = low.2
	title = low.2.t
	desc = low.2.d
	picture = GFX_report_event_german_parade_paris
	
	fire_only_once = yes
	
	trigger = {
		tag = BEL
		has_war_with = GER
		6 = { is_controlled_by = GER }
		is_in_faction_with = ENG
		ENG = { has_war_with = GER }
		surrender_progress > 0.3
		GER = {
			is_faction_leader = yes
		}
		NOT = {
			has_civil_war = yes
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	immediate = {

	}
	
	option = { # Become Belgian Government in Exile
		name = low.2.a
		ai_chance = {
			factor = 85
		}
		#Set same cosmetic tag they would get on capitulation
		BEL = {
			set_capital = 295 
			add_stability = -0.10
			add_war_support = 0.10
			set_cosmetic_tag = BEL_EXILE
			#belgian congo
			annex_country = { target = COG transfer_troops = yes }
			hidden_effect = {
				set_state_controller = 295
				set_state_controller = 793
				set_state_controller = 792
				set_state_controller = 791
				set_state_controller = 718
				set_state_controller = 538
				set_state_controller = 940
			}
		}
		COG = { custom_effect_tooltip = GAME_OVER_TT }
	}
	option = { # Surrender
		name = low.2.b
		ai_chance = {
			factor = 15
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		
		custom_effect_tooltip = GAME_OVER_TT
	}
}

### Belgium is restored

# Give United States Dutch Suriname
country_event = {
	id = low.6
	title = low.6.t
	desc = low.6.d
	picture = GFX_report_event_generic_peaceful_annexation
	
	fire_only_once = yes
	
	trigger = {
		tag = HOL
		has_war_with = GER
		7 = { is_controlled_by = GER }
		is_in_faction_with = ENG
		ENG = { has_war_with = GER }
		GER = {
			is_faction_leader = yes
		}
		NOT = {
			has_civil_war = yes
		}
		date > 1941.11.20
		date < 1941.11.25
	}
	
	mean_time_to_happen = {
		days = 1
	}

	immediate = {

	}
	
	option = { # Give United States Dutch Suriname temporarily
		name = low.6.a
		ai_chance = {
			factor = 85
		}
		#Set same cosmetic tag they would get on capitulation
		HOL = {
			release_puppet = SUR
		}
		USA = {
			puppet = SUR
			set_autonomy = { target = SUR autonomy_state = autonomy_colony }
		}
	}
	option = { # Keep Dutch Suriname
		name = low.6.b
		ai_chance = {
			factor = 15
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
	}
}

	}

	immediate = {

	}
	
	option = { # United States returns Dutch Suriname
		name = low.7.a

			factor = 65
		}

		}
	}
	option = { # United States refuses to return Dutch Suriname
		name = low.7.b

			factor = 45
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		HOL = {
			add_opinion_modifier = { 
				target = USA 
				modifier = US_refuses_to_return_suriname
			}
		}
	}
}
