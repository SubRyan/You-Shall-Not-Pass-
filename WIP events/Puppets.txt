### Nigeria is free

# Nigeria
country_event = {
	id = puppet.1
	title = puppet.1.t
	desc = puppet.1.d
	picture = GFX_report_event_generic_sign_treaty1
	
	is_triggered_only = yes
	
	trigger = {
		NIG = {
			AND ={
				NOT ={
					is_subject_of = ENG
				}
				exists = yes
				is_puppet = no
			}
		}
	}
	immediate ={
		NIG = {
			drop_cosmetic_tag = yes
		}
	}
	option = {
		name = puppet.1.a
		NIG = {
			add_political_power = 50
		}
	}
}

	desc = puppet.107.d

	is_triggered_only = yes
	
	trigger = {
		INS = {
			AND ={
				NOT ={
					is_subject_of = HOL
				}

				is_puppet = no
			}
		}

	immediate ={
		INS = {
			drop_cosmetic_tag = yes
		}
	}
	option = {
		name = puppet.107.a
		INS = {
			add_political_power = 50
		}
	}
}
